== API Endpoints Summary

[cols="1,2,2,1",options="header"]
|===
| Method | Endpoint | Purpose | Status Code
| GET | /api/schedule/matches | Fetch all matches | 200
| POST | /api/schedule/matches | Create new match | 201
| PUT | /api/schedule/matches/\{id} | Update match details | 200
| DELETE | /api/schedule/matches/\{id} | Delete match | 200
| PUT | /api/schedule/matches/\{id}/status | Change match status | 200
| POST | /api/schedule/matches/\{id}/result | Set match result | 200
| POST | /api/schedule/export | Export schedule as JSON | 200
|===

=== Step 2: Create Schedule Service

Create a new service file for schedule operations:

[source,javascript]
----
// services/scheduleService.js
class ScheduleService {
  constructor() {
    this.baseUrl = API_BASE_URL;
  }

  // GET /api/schedule/matches
  async getAllMatches() {
    const response = await fetch(`${this.baseUrl}/api/schedule/matches`);
    const data = await response.json();
    if (!data.success) throw new Error(data.message);
    return data.data;
  }

  // POST /api/schedule/matches
  async createMatch(matchData) {
    const response = await fetch(`${this.baseUrl}/api/schedule/matches`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(matchData)
    });
    const data = await response.json();
    if (!data.success) throw new Error(data.message);
    return data.data;
  }

  // PUT /api/schedule/matches/{id}
  async updateMatch(id, matchData) {
    const response = await fetch(`${this.baseUrl}/api/schedule/matches/${id}`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(matchData)
    });
    const data = await response.json();
    if (!data.success) throw new Error(data.message);
    return data.data;
  }

  // DELETE /api/schedule/matches/{id}
  async deleteMatch(id) {
    const response = await fetch(`${this.baseUrl}/api/schedule/matches/${id}`, {
      method: 'DELETE'
    });
    const data = await response.json();
    if (!data.success) throw new Error(data.message);
    return data.data;
  }

  // PUT /api/schedule/matches/{id}/status
  async updateMatchStatus(id, status) {
    const response = await fetch(`${this.baseUrl}/api/schedule/matches/${id}/status`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ status })
    });
    const data = await response.json();
    if (!data.success) throw new Error(data.message);
    return data.data;
  }

  // POST /api/schedule/matches/{id}/result
  async setMatchResult(id, resultData) {
    const response = await fetch(`${this.baseUrl}/api/schedule/matches/${id}/result`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(resultData)
    });
    const data = await response.json();
    if (!data.success) throw new Error(data.message);
    return data.data;
  }

  // POST /api/schedule/export
  async exportSchedule() {
    const response = await fetch(`${this.baseUrl}/api/schedule/export`, {
      method: 'POST'
    });
    const data = await response.json();
    if (!data.success) throw new Error(data.message);
    return data.data;
  }
}

export default new ScheduleService();

// components/MatchList.js
import React, { useState, useEffect } from 'react';
import ScheduleService from '../services/scheduleService';

const MatchList = () => {
  const [matches, setMatches] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    loadMatches();
  }, []);

  const loadMatches = async () => {
    try {
      setLoading(true);
      const data = await ScheduleService.getAllMatches();
      setMatches(data);
      setError(null);
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };

  const getStatusBadge = (status) => {
    const badges = {
      scheduled: 'badge-secondary',
      live: 'badge-success',
      completed: 'badge-primary'
    };
    return badges[status] || 'badge-secondary';
  };

  if (loading) return <div className="text-center">Loading matches...</div>;
  if (error) return <div className="alert alert-danger">{error}</div>;

  return (
    <div className="match-list">
      <h2>Cricket Schedule</h2>
      {matches.length === 0 ? (
        <p>No matches scheduled yet.</p>
      ) : (
        <div className="row">
          {matches.map(match => (
            <div key={match.id} className="col-md-6 mb-3">
              <div className="card">
                <div className="card-header">
                  <h5>{match.round} - Match {match.match_number}</h5>
                  <span className={`badge ${getStatusBadge(match.status)}`}>
                    {match.status.toUpperCase()}
                  </span>
                </div>
                <div className="card-body">
                  <div className="d-flex justify-content-between align-items-center mb-2">
                    <span>{match.team1}</span>
                    <span className="text-muted">vs</span>
                    <span>{match.team2}</span>
                  </div>

                  {match.status === 'completed' && match.result && (
                    <div className="alert alert-info">
                      <strong>Result:</strong> {match.result.winner} won by{' '}
                      {match.result.margin} {match.result.margin_type}
                      {match.result.won_by_batting_first && ' (batting first)'}
                    </div>
                  )}
                </div>
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
};

// components/MatchResultForm.js
import React, { useState } from 'react';
import ScheduleService from '../services/scheduleService';

const MatchResultForm = ({ match, onResultSet, onClose }) => {
  const [resultData, setResultData] = useState({
    winner: '',
    margin: '',
    marginType: 'runs',
    wonByBattingFirst: false
  });
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const handleSubmit = async (e) => {
    e.preventDefault();
    setLoading(true);
    setError(null);

    try {
      const data = {
        ...resultData,
        margin: parseInt(resultData.margin)
      };

      const updatedMatch = await ScheduleService.setMatchResult(match.id, data);
      onResultSet(updatedMatch);
      onClose();
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };

  const handleChange = (e) => {
    const { name, value, type, checked } = e.target;
    setResultData({
      ...resultData,
      [name]: type === 'checkbox' ? checked : value
    });
  };

  return (
    <div className="modal show d-block" tabIndex="-1">
      <div className="modal-dialog">
        <div className="modal-content">
          <div className="modal-header">
            <h5 className="modal-title">Set Match Result</h5>
            <button type="button" className="btn-close" onClick={onClose}></button>
          </div>
          <form onSubmit={handleSubmit}>
            <div className="modal-body">
              {error && <div className="alert alert-danger">{error}</div>}

              <div className="mb-3">
                <label className="form-label">Winner</label>
                <select
                  className="form-select"
                  name="winner"
                  value={resultData.winner}
                  onChange={handleChange}
                  required
                >
                  <option value="">Select Winner</option>
                  <option value={match.team1}>{match.team1}</option>
                  <option value={match.team2}>{match.team2}</option>
                </select>
              </div>

              <div className="row">
                <div className="col-md-6 mb-3">
                  <label className="form-label">Margin</label>

                    type="number"
                    className="form-control"
                    name="margin"
                    value={resultData.margin}
                    onChange={handleChange}

                    min="1"

                  <select
                    className="form-select"

// pages/AdminSchedule.js
import React, { useState, useEffect } from 'react';
import ScheduleService from '../services/scheduleService';
import MatchForm from '../components/MatchForm';
import MatchResultForm from '../components/MatchResultForm';

const AdminSchedule = () => {
  const [matches, setMatches] = useState([]);
  const [showCreateForm, setShowCreateForm] = useState(false);
  const [selectedMatch, setSelectedMatch] = useState(null);
  const [showResultForm, setShowResultForm] = useState(false);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadMatches();
  }, []);

  const loadMatches = async () => {
    try {
      const data = await ScheduleService.getAllMatches();
      setMatches(data);
    } catch (err) {
      alert('Failed to load matches: ' + err.message);
    } finally {
      setLoading(false);
    }
  };

  const handleMatchCreated = (newMatch) => {
    setMatches([...matches, newMatch]);
    setShowCreateForm(false);
  };

  const handleStatusChange = async (matchId, newStatus) => {
    try {
      await ScheduleService.updateMatchStatus(matchId, newStatus);
      loadMatches(); // Refresh list
    } catch (err) {
      alert('Failed to update status: ' + err.message);
    }
  };

  const handleDeleteMatch = async (matchId) => {
    if (!confirm('Are you sure you want to delete this match?')) return;

    try {
      await ScheduleService.deleteMatch(matchId);
      setMatches(matches.filter(m => m.id !== matchId));
    } catch (err) {
      alert('Failed to delete match: ' + err.message);
    }
  };

  const handleResultSet = (updatedMatch) => {
    setMatches(matches.map(m => m.id === updatedMatch.id ? updatedMatch : m));
    setShowResultForm(false);
    setSelectedMatch(null);
  };

  if (loading) return <div className="text-center">Loading...</div>;

  return (
    <div className="admin-schedule">
      <div className="d-flex justify-content-between align-items-center mb-4">
        <h1>Schedule Management</h1>
        <button
          className="btn btn-primary"
          onClick={() => setShowCreateForm(true)}
        >
          Create Match
        </button>
      </div>

      <div className="table-responsive">
        <table className="table table-striped">
          <thead>
            <tr>
              <th>Round</th>
              <th>Match</th>
              <th>Teams</th>
              <th>Status</th>
              <th>Actions</th>
            </tr>
          </thead>
          <tbody>
            {matches.map(match => (
              <tr key={match.id}>
                <td>{match.round} ({match.round_number})</td>
                <td>{match.match_number}</td>
                <td>{match.team1} vs {match.team2}</td>
                <td>
                  <span className={`badge bg-${match.status === 'live' ? 'success' : match.status === 'completed' ? 'primary' : 'secondary'}`}>
                    {match.status}
                  </span>
                </td>
                <td>
                  {match.status === 'scheduled' && (
                    <>
                      <button
                        className="btn btn-sm btn-success me-2"
                        onClick={() => handleStatusChange(match.id, 'live')}
                      >
                        Start
                      </button>
                      <button
                        className="btn btn-sm btn-danger me-2"
                        onClick={() => handleDeleteMatch(match.id)}
                      >
                        Delete
                      </button>
                    </>
                  )}
                  {match.status === 'live' && (
                    <>
                      <button
                        className="btn btn-sm btn-primary me-2"
                        onClick={() => {
                          setSelectedMatch(match);
                          setShowResultForm(true);
                        }}
                      >
                        Set Result
                      </button>
                      <button
                        className="btn btn-sm btn-warning"
                        onClick={() => handleStatusChange(match.id, 'scheduled')}
                      >
                        Pause
                      </button>
                    </>
                  )}
                  {match.status === 'completed' && match.result && (
                    <span className="text-success">
                      {match.result.winner} won by {match.result.margin} {match.result.margin_type}
                    </span>
                  )}
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      {showCreateForm && (
        <MatchForm
          onMatchCreated={handleMatchCreated}
          onClose={() => setShowCreateForm(false)}
        />
      )}

      {showResultForm && selectedMatch && (
        <MatchResultForm
          match={selectedMatch}
          onResultSet={handleResultSet}
          onClose={() => {
            setShowResultForm(false);
            setSelectedMatch(null);
          }}
        />
      )}
    </div>
  );
};

== Troubleshooting

=== Common Issues

==== "Team not found" Error
**Problem**: Team name doesn't exist in database +
**Solution**: Use exact team names from `/api/teams` endpoint

==== "Cannot update completed match" Error
**Problem**: Trying to modify a match with status 'completed' +
**Solution**: Only scheduled/live matches can be modified

==== "Invalid margin" Error
**Problem**: Margin value outside cricket rules +
**Solution**: Runs margin: 1-999, Wickets margin: 1-10

==== CORS Errors
**Problem**: Frontend can't connect to backend +
**Solution**: Add frontend domain to backend CORS settings

==== 404 Errors
**Problem**: API endpoints not found +
**Solution**: Ensure backend is deployed with schedule routes
